class Node
{
}

enum LiteralValue
{
    True,
    False,
    Null,
}

class Literal : Node
{
    var value : LiteralValue;
}

class String : Node
{
    var content : token;
}

class Number : Node
{
    var content : token;
}

class Array : Node
{
    var items : Node[];
}

class ObjectField
{
    var name : token;
    var value : Node;
}

class Object : Node
{
    var fields : ObjectField[];
}
