Do
  ::= "do" ";" as DoStat
  ;

Block
  ::= "{" {Stat:stats} "}" as BlockStat
  ;

IfTail_SWITCH_0allow_half_if
  ::= OtherStat:thenBranch "else" Stat:elseBranch as partial IfStat
  ::= If:thenBranch "else" Stat:elseBranch as partial IfStat
  ;

IfTail_SWITCH_1allow_half_if
  ::= OtherStat:thenBranch "else" Stat:elseBranch as partial IfStat
  ::= If:thenBranch "else" Stat:elseBranch as partial IfStat
  ::= If:thenBranch as partial IfStat
  ;

If_SWITCH_0allow_half_if
  ::= "if" "cond" "then" IfTail as IfStat
  ;

If_SWITCH_1allow_half_if
  ::= "if" "cond" "then" IfTail as IfStat
  ;

OtherStat
  ::= !Do
  ::= !Block
  ;

Stat
  ::= !OtherStat
  ::= !If
  ;

Module
  ::= Stat:stat as Module
  ;
