//////////////////////////////////////////////////////////////////
// Static Assert Declaration
//////////////////////////////////////////////////////////////////

_StaticAssertDecl
  ::= "static_assert" "(" _Expr:expr ["," _Expr:message] ")" as StaticAssertDeclaration
  ;
  
//////////////////////////////////////////////////////////////////
// Typedef Declaration
//////////////////////////////////////////////////////////////////

_TypedefCompatibleDecl
  ::= !_MultiVarsDeclWithoutKeyword
  ::= !_ClassDecl
  ::= !_EnumDecl
  ;

_TypedefDecl
  ::= "typedef" _TypedefCompatibleDecl:decl as TypedefDeclaration
  ;
  
//////////////////////////////////////////////////////////////////
// Extern Declaration
//////////////////////////////////////////////////////////////////

_ExternDecl
  ::= "extern" [STRING] "{" {_DeclWithSemicolon:decls} "}" as ExternDeclaration
  ;
  
//////////////////////////////////////////////////////////////////
// Namespace Declaration
//////////////////////////////////////////////////////////////////

_NsName
  ::= ID:name as NamespaceName
  ;

_NsDecl
  ::= "namespace" {_NsName:names ; "::"} "{" {_DeclWithSemicolon:decls} "}" as NamespaceDeclaration
  ;

//////////////////////////////////////////////////////////////////
// Enum Declaration
//////////////////////////////////////////////////////////////////

_EnumItem
  ::= ID:name ["=" _Expr:expr] as EnumItem
  ;

_EnumBody
  ::= "{" [_EnumItem:items {_EnumItem:items ; ","} [","]] "}" [_MultiVarsDeclVariablePart:firstVarPart] as EnumBody
  ;

_EnumDecl
  ::= "enum" ["class"] [ID:name] _EnumBody:body as EnumDeclaration {kind = Enum}
  ;

_EnumForwardDecl
  ::= "enum" ["class"] ID:name as EnumDeclaration {kind = Enum}
  ;
