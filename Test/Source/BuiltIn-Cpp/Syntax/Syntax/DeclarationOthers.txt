//////////////////////////////////////////////////////////////////
// Namespace Declarations
//////////////////////////////////////////////////////////////////

_NsName
  ::= ID:name as NamespaceName
  ;

_NsDecl
  ::= "namespace" {ID:names ; "::"} "{" _Decl:decls "}" as NamespaceDeclaration
  ;

//////////////////////////////////////////////////////////////////
// Other Declarations
//////////////////////////////////////////////////////////////////

_StaticAssertDecl
  ::= "static_assert" "(" _Expr:expr ["," _Expr:message] ")" as StaticAssertDeclaration
  ;

_OtherDecl
  ::= !_StaticAssertDecl
  ::= !_NsDecl
  ;
