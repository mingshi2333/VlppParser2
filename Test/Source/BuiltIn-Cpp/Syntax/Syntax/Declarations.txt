//////////////////////////////////////////////////////////////////
// Typedef Declarations
//////////////////////////////////////////////////////////////////

_TypedefCompatibleDecl
  ::= !_VarsDeclWithoutKeyword
  ::= !_ClassDeclInTypedef
  ;

_TypedefDecl
  ::= "typedef" _TypedefCompatibleDecl:declaration as TypedefDeclaration
  ;
  
//////////////////////////////////////////////////////////////////
// Declarations
//////////////////////////////////////////////////////////////////

_Decl
  ::= !_VarsDecl
  ::= !_TypedefDecl
  ::= !_OtherDecl
  ;

_DeclWithSemicolon
  ::= !_Decl ";"
  ;
