Do
  ::= "do" ";" as DoStat
  ;

Block
  ::= "{" { Stat:stats } "}" as BlockStat
  ;

IfThenStat
  ::= "if" "cond" "then" Stat:thenBranch as IfContentCandidate
  ;

IfContent
  ::= !IfThenStat
  ::= !IfThenStat "else" Stat:elseBranch
  ;

Stat
  ::= !Do
  ::= !Block
  ::= IfContent:content as IfStat
  ;

Module
  ::= Stat:stat as Module
  ;
