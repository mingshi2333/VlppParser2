switch allow_half_if;

Do
  ::= "do" ";" as DoStat
  ;

Block
  ::= "{" { Stat:stats } "}" as BlockStat
  ;

IfTail
  ::= ?(allow_half_if: Stat:thenBranch) as partial IfStat
  ::= !(!allow_half_if; Stat:thenBranch) "else" Stat:elseBranch as partial IfStat
  ;

If
  ::= "if" "cond" "then" IfTail as IfStat
  ;

Stat
  ::= !Do
  ::= !(allow_half_if; !Block)
  ::= !If
  ;

Module
  ::= Stat:stat as Module
  ;
