////////////////////////////////////////
// Left Recursive (Primitives)
////////////////////////////////////////

_Name
  ::= ID:name as Name
  ::= _Name:parent "." ID:member as MemberName
  ;

_PrimitiveShared
  ::= !_Name
  ;

_PrimitiveTypeOnly
  ::= "const" _PrimitiveShared:type as ConstType
  ::= "const" _PrimitiveTypeOnly:type as ConstType
  ;

_PrimitiveExprOnly
  ::= "(" !_Expr ")"
  ;

////////////////////////////////////////
// Left Recursive (Types)
////////////////////////////////////////

_LongType
  ::= left_recursion_placeholder(LRI__LongType)
  ::= !_PrimitiveShared
  ::= !_PrimitiveTypeOnly
  ::= _LongType:type "const" as ConstType
  ::= _LongType:type "*" as PointerType
  ::= _LongType:returnType "(" {_LongType:args ; ","} ")" as FunctionType
  ;

////////////////////////////////////////
// Left Recursive (Expressions)
////////////////////////////////////////

_Expr0
  ::= left_recursion_placeholder(LRI__Expr)
  ::= !_PrimitiveShared
  ::= !_PrimitiveExprOnly
  ::= _LongType:type "{" {_Expr:args ; ","} "}" as CtorExpr
  ;

_Expr1_LRI_Original
  ::= !_Expr0
  ::= _Expr1_LRI_Original:func "(" {_Expr:args ; ","} ")" as CallExpr
  ;

_Expr_LRI_Original
  ::= !_Expr1_LRI_Original
  ::= _Expr_LRI_Original:first "*" _Expr1:second as MulExpr
  ;

////////////////////////////////////////
// Left Recursive (Module)
////////////////////////////////////////

Module_LRI_Original
  ::= !_Expr
  ;

////////////////////////////////////////
// Left Recursive (Rewritten)
////////////////////////////////////////

_Expr1
  ::= !_PrimitiveShared [left_recursion_inject_multiple(LRI__Expr) _Expr1_LRI_Original]
  ::= !_PrimitiveShared left_recursion_inject(LRI__LongType) _Expr1_LRI_Original
  ::= !_PrimitiveTypeOnly left_recursion_inject(LRI__LongType) _Expr1_LRI_Original
  ::= !_PrimitiveExprOnly [left_recursion_inject(LRI__Expr) _Expr1_LRI_Original]
  ;

_Expr
  ::= !_PrimitiveShared [left_recursion_inject_multiple(LRI__Expr) _Expr_LRI_Original]
  ::= !_PrimitiveShared left_recursion_inject(LRI__LongType) _Expr_LRI_Original
  ::= !_PrimitiveTypeOnly left_recursion_inject(LRI__LongType) _Expr_LRI_Original
  ::= !_PrimitiveExprOnly [left_recursion_inject(LRI__Expr) _Expr_LRI_Original]
  ;

Module
  ::= !_PrimitiveShared [left_recursion_inject_multiple(LRI__Expr) _Expr_LRI_Original]
  ::= !_PrimitiveShared left_recursion_inject(LRI__LongType) _Expr_LRI_Original | _LongType
  ::= !_PrimitiveTypeOnly [left_recursion_inject(LRI__LongType) _Expr_LRI_Original | _LongType]
  ::= !_PrimitiveExprOnly [left_recursion_inject(LRI__Expr) _Expr_LRI_Original]
  ;
